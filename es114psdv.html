import matplotlib.pyplot as plt

def fibonacci(n):
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]

    fib_sequence = [0, 1]
    for i in range(2, n):
        next_fib = fib_sequence[-1] + fib_sequence[-2]
        fib_sequence.append(next_fib)
    
    return fib_sequence


def plot_fibonacci(n):
    fib_sequence = fibonacci(n)
    
    plt.figure(figsize=(12, 6)) 
    plt.plot(fib_sequence, marker='o', linestyle='-', color='skyblue', markersize=8, linewidth=2)  # Enhanced line style
    
    
    plt.title(f'First {n} Fibonacci Numbers', fontsize=18)
    plt.xlabel('Index', fontsize=14)
    plt.ylabel('Fibonacci Number', fontsize=14)
    plt.grid(color='gray', linestyle='--', alpha=0.7)
    plt.xticks(range(n), fontsize=12)  
    plt.yticks(fontsize=12)

  
    plt.tight_layout()  
    plt.show()

    def isPalindrome(self, s: str) -> bool:
        l, r = 0, len(s) - 1

        while l < r:
            while l < r and not self.alphaNum(s[l]):
                l += 1
            while r > l and not self.alphaNum(s[r]):
                r -= 1

            if s[l].lower() != s[r].lower():
                return False

            l, r = l + 1, r - 1

        return True

    def alphaNum(self, c):
        return (ord('A') <= ord(c) <= ord('Z') or 
                ord('a') <= ord(c) <= ord('z') or 
                ord('0') <= ord(c) <= ord('9'))

from statistics import mean, median

def remove_duplicates_and_calculate_stats(input_list):
    unique_list = []
    for item in input_list:
        if item not in unique_list:
            unique_list.append(item)
    
    if unique_list:  
        calculated_mean = mean(unique_list)
        calculated_median = median(unique_list)
    else:
        calculated_mean = None
        calculated_median = None
    
    return unique_list, calculated_mean, calculated_median

    return [w for w in words if sorted(w) == sorted_word]

import matplotlib.pyplot as plt
from collections import Counter
import re

def analyze_text_file(file_path):

    try:
        
        with open(file_path, 'r', encoding='utf-8') as file:
            text = file.read().lower()
        
       
        words = re.findall(r'\b\w+\b', text)
        
    
        word_counts = Counter(words)
        
     
        top_words = word_counts.most_common(10)
        
        # Visualization
        plt.figure(figsize=(12, 6))
        
        # Bar chart of word frequencies
        plt.subplot(121)
        words, frequencies = zip(*top_words)
        plt.bar(words, frequencies, color='skyblue')
        plt.title('Top 10 Word Frequencies')
        plt.xlabel('Words')
        plt.ylabel('Frequency')
        plt.xticks(rotation=45)
        
        
        
        plt.tight_layout()
        plt.show()
        
        # Return analysis results
        return {
            'total_words': len(words),
            'unique_words': len(set(words)),
            'most_frequent_word': word_counts.most_common(1)[0],
            'word_frequencies': dict(top_words)
        }
    
    except FileNotFoundError:
        print(f"File not found: {file_path}")
        return None
    except Exception as e:
        print(f"Error processing file: {e}")
        return None

# Example usage

import numpy as np
import matplotlib.pyplot as plt
import statistics
from scipy import stats

def advanced_statistical_analysis(numbers):
    
   
    mean_value = np.mean(numbers)
    median_value = np.median(numbers)
    
    try:
        mode_value = statistics.mode(numbers)
    except statistics.StatisticsError:
        mode_value = "No unique mode"
    
    std_dev = np.std(numbers)
    variance = np.var(numbers)
    
    q1 = np.percentile(numbers, 25)
    q3 = np.percentile(numbers, 75)
    iqr = q3 - q1
    
    # Visualization
    plt.figure(figsize=(12, 5))
    
    # Box Plot
    plt.subplot(121)
    plt.boxplot(numbers, vert=True)
    plt.title('Data Distribution Boxplot')
    plt.ylabel('Values')
    
    
    
    plt.tight_layout()
    plt.show()
    
    # Outlier Detection
    z_scores = np.abs(stats.zscore(numbers))
    outliers = [num for num, z in zip(numbers, z_scores) if z > 3]
    
    return {
        'mean': mean_value,
        'median': median_value,
        'mode': mode_value,
        'standard_deviation': std_dev,
        'variance': variance,
        'quartile_1': q1,
        'quartile_3': q3,
        'interquartile_range': iqr,
        'outliers': outliers
    }

# Example Usage
sample_data = [10, 15, 20, 25, 30, 35, 40, 45, 50, 22, 18, 200]
result = advanced_statistical_analysis(sample_data)
print(result)

{'mean': 42.5, 'median': 27.5, 'mode': 10, 'standard_deviation': 48.932777017727766, 'variance': 2394.4166666666665, 'quartile_1': 19.5, 'quartile_3': 41.25, 'interquartile_range': 21.75, 'outliers': [200]}

import numpy as np
import matplotlib.pyplot as plt


def normalize_data(numbers, method='minmax'):
  
  
    if method == 'minmax':
        normalized = (numbers - np.min(numbers)) / (np.max(numbers) - np.min(numbers))
    
 
    else:
        raise ValueError("Invalid normalization method")
    
    # Visualization
    plt.figure(figsize=(12, 5))
    
    # Original vs Normalized Line Plot
    plt.subplot(121)
    plt.plot(numbers, label='Original', marker='o')
    plt.plot(normalized, label='Normalized', marker='x')
    plt.title(f'{method.capitalize()} Normalization')
    plt.legend()
    
    
    
    plt.tight_layout()
    plt.show()
    
    return {
        'original_data': numbers,
        'normalized_data': normalized,
        'min_value': np.min(normalized),
        'max_value': np.max(normalized),
        'method': method
    }

# Example Usage
sample_data = [27, 10, 45, 45, 19, 34, 40, 40, 5]

# Different Normalization Methods
print("Min-Max Normalization:")
minmax_result = normalize_data(sample_data, method='minmax')
print(minmax_result)
